        for(int x=0; x<=3; x++) {
            for (int y=0; y<=x; y++) {
                for (int z=0; z<=y; z++) {
                    if (x==0 && y==0 && z==0) {
                        continue;
                    }
                    pBoardNum++;
                    System.out.println("Possible Board Number " + pBoardNum);

                    reducer(x, y, z);
                    //Create all the boards and run them through the reducer
                }
            }

    }


    public void reducer(int x, int y, int z) {
        Losing.add(new Board(1,0,0));
        System.out.println("Reduced Boards: ");
        int i=x, j=y, k=z;

                System.out.println("game over");
                break;
            }
            for (Board board : Losing) {

                if (i == board.col0 && j == board.col1 && k == board.col2) {
                    Winning.add(new Board(x, y, z));
                    isLosing = true;
                }
            }
            if(!isLosing){
                Losing.add(new Board(x,y,z));
            }
            System.out.println(i+","+j+","+k);
        }

        k=z;
        while(j>0){
            j--;
            if(j<k) {
                k = j;
            }else if(i==0 && j==0 && k==0){

            }
            for (Board board : Losing) {

                    isLosing = true;
                }
            }
            if(!isLosing){
                Losing.add(new Board(x,y,z));
            }
            System.out.println(i+","+j+","+k);
        }
        j=y;
        k=z;
        isLosing=false;
        while (i>0){
            i--;
            if (i < j && i < k) {
                j = i;
                k = i;
            } else if (i < j) {
                j = i;
            } else if (i < k) {
                k = i;
            } if(i==0 && j==0 && k==0){
                break;

            for (Board board : Losing) {
                if (x == board.col0 && y == board.col1 && z == board.col2) {
                    Winning.add(new Board(x,y,z));
                    isLosing = true;
